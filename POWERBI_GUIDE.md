## 2. DAX MEASURES

### KPI Measures

```dax
Total Customers =
COUNTROWS(customers)

Churned Customers =
CALCULATE(COUNTROWS(customers), customers[Churn] = 1)

Churn Rate % =
DIVIDE([Churned Customers], [Total Customers]) * 100

Retained Customers =
[Total Customers] - [Churned Customers]

Avg Monthly Charges =
AVERAGE(customers[MonthlyCharges])

Avg Tenure (Months) =
AVERAGE(customers[tenure])

At-Risk Customers =
CALCULATE(
    COUNTROWS(customers),
    customers[Contract] = "Month-to-month",
    customers[tenure] < 12
)

Projected Churn Reduction % =
0.23
```

### Contract-Based Analysis

```dax
Month-to-Month Churn Rate =
CALCULATE(
    DIVIDE(
        CALCULATE(COUNTROWS(customers), customers[Churn] = 1),
        COUNTROWS(customers)
    ) * 100,
    customers[Contract] = "Month-to-month"
)

One-Year Churn Rate =
CALCULATE(
    DIVIDE(
        CALCULATE(COUNTROWS(customers), customers[Churn] = 1),
        COUNTROWS(customers)
    ) * 100,
    customers[Contract] = "One year"
)

Two-Year Churn Rate =
CALCULATE(
    DIVIDE(
        CALCULATE(COUNTROWS(customers), customers[Churn] = 1),
        COUNTROWS(customers)
    ) * 100,
    customers[Contract] = "Two year"
)
```

### Trend Measures (Month-on-Month simulation)

```dax
Tenure Bucket =
SWITCH(
    TRUE(),
    customers[tenure] <= 6,   "0-6 months",
    customers[tenure] <= 12,  "7-12 months",
    customers[tenure] <= 24,  "1-2 years",
    customers[tenure] <= 48,  "2-4 years",
    "4+ years"
)

Churn Rate by Tenure =
DIVIDE(
    CALCULATE(COUNTROWS(customers), customers[Churn] = 1),
    COUNTROWS(customers)
) * 100

Revenue at Risk =
CALCULATE(
    SUMX(customers, customers[MonthlyCharges]),
    customers[Churn] = 1
)

Projected Revenue Saved =
[Revenue at Risk] * [Projected Churn Reduction %]
```

### Model Performance Measures

```dax
XGBoost Recall =
CALCULATE(
    MAX(model_comparison[Recall]),
    model_comparison[Model] = "XGBoost"
)

Best Model =
CALCULATE(
    FIRSTNONBLANK(model_comparison[Model], 1),
    TOPN(1, model_comparison, model_comparison[F1 Score], DESC)
)
```
